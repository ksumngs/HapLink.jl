Getting started

Ready to dive in? ðŸ¤¿ Here's a 30,000-foot view

curl -fsSL https://install.julialang.org | sh -s -- --yes
julia \
  --startup-file=no \
  --history-file=no \
  --project=$(mktemp -d) \
  --quiet \
  -e 'using Pkg; Pkg.add(HapLink)'
echo 'export PATH=$HOME/.julia/bin:$PATH' >> $HOME/.bashrc
source ~/.bashrc

git clone https://github.com/ksumngs/HapLink.jl
cd HapLink.jl/example

haplink variants \
  reference.fasta \
  sample.bam \
  --significance 0.5 \
  --depth 1 \
  --quality 10.0 \
  --position 0.01 \
  --frequency 0.05 \
  | tee sample.vcf

haplink consensus \
  reference.fasta \
  sample.vcf \
  | tee consensus.fasta

haplink haplotypes \
  reference.fasta \
  sample.vcf \
  sample.bam \
  --frequency 0.75 \
  | tee sample.yml

haplink sequences \
  reference.fasta \
  sample.yml \
  | tee sample.fasta
